# title, startup log) all derive from env!("CARGO_PKG_VERSION"), so bumping
# Cargo.toml updates them automatically — the doc files below must be matched
# by hand.
#
# Current version: 0.6.1
#

Cargo.toml              # package version (line 3 only — NOT the russh/dep "0.60" specs); Cargo.lock follows on build
versionchange.txt       # the "Current version:" line in this file's header

web/index.html          # <title>, .hdr-version, startup-banner output-text, footer
web/appendix.html       # .hdr-version, footer
web/kermit.html         # <title>, .hdr-version, "Ethernet Gateway Kermit X.Y.Z" identity strings, footer
